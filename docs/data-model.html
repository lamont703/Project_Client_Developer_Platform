Data Model Additions

  • Developer:
    • status: "pending" | "approved" | "rejected" | "waitlisted"
    • tier: "Rising" | "Pro" | "Elite"
    • vettingNotes: string[]
    • portfolio: {title, url, tech[]}[]
  • TalentPool (singleton):
    • maxSlots: number
    • currentApproved: number
    • waitlistCount: number
    • monthlyIntakeLimit: number
  • Application:
    • stage: "submitted" | "shortlisted" | "declined"
  • Job:
    • publicStatus: "posted" | "shortlisting" | "hired" | "in_progress" | "prototype_ready" | "completed"
    • statusHistory: {status, atISO, note?}[]
    • isPublicStatusEnabled: boolean (default true)
    • visibility: "public" | "private"
